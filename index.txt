const emojiMap={Flood:'🌊',Earthquake:'🏔️',Drought:'☀️',Storm:'🌪️'};
async function predict(){
  const btn=document.querySelector('.pred-btn');
  btn.textContent='⏳ Analyzing...';
  btn.disabled=true;
  try{
    const res=await fetch('/predict',{
      method:'POST',
      headers:{'Content-Type':'application/json'},
      body:JSON.stringify({
        year:document.getElementById('year').value,
        month:document.getElementById('month').value,
        deaths:document.getElementById('deaths').value,
        affected:document.getElementById('affected').value
      })
    });
    const data=await res.json();
    const probs=data.probabilities;

    // Result card update
    const box=document.getElementById('result');
    box.style.display='block';
    document.getElementById('res-icon').textContent=emojiMap[data.disaster]||'⚠️';
    document.getElementById('res-type').textContent=data.disaster;
    document.getElementById('res-conf').textContent=`Confidence: ${data.confidence}%`;
    setTimeout(()=>document.getElementById('res-fill').style.width=data.confidence+'%',100);

    // Bar chart update
    if(window.barChartRef){
      window.barChartRef.data.datasets[0].data=[
        probs['Flood']||0,
        probs['Storm']||0,
        probs['Earthquake']||0,
        probs['Drought']||0
      ];
      window.barChartRef.update();
    }

    // Pie chart update
    if(window.pieChartRef){
      window.pieChartRef.data.datasets[0].data=[
        probs['Flood']||0,
        probs['Storm']||0,
        probs['Earthquake']||0,
        probs['Drought']||0
      ];
      window.pieChartRef.update();
    }

    // Map markers highlight
    window.highlightMapMarkers(data.disaster);

    // Data section pe scroll
    setTimeout(()=>document.getElementById('data').scrollIntoView({behavior:'smooth'}),800);

  }catch(e){alert('Error! Flask server chal raha hai?');}
  btn.textContent='⚡ PREDICT DISASTER TYPE';
  btn.disabled=false;
}
